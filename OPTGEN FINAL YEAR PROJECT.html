Python Client Example
import requests, time # 1. Upload CAD file with open("bracket.step", "rb") as f: r = requests.post("http://localhost:8000/upload-cad", files={"file": f}) cad = r.json() file_id = cad["file_id"] dims = cad["dimensions"] # 2. Launch GA optimization payload = { "cad_file_id": file_id, "load_n": 5000, "max_stress_mpa": 250, "max_weight_kg": 5.0, "objective": "balanced", "materials": ["Aluminum_6061", "Titanium_Ti64"], "ga_params": {"population_size": 50, "generations": 100} } r = requests.post("http://localhost:8000/generate-designs", json=payload) job_id = r.json()["job_id"] # 3. Poll until complete while True: r = requests.get(f"http://localhost:8000/results?job_id={job_id}") d = r.json() if d["status"] == "complete": break time.sleep(2) top = d["result"]["top_designs"][0] # 4. Human-in-the-loop selection requests.post("http://localhost:8000/select-design", json={ "job_id": job_id, "design_id": top["design_id"], "approved_by": "eng_alice", "user_notes": "Preferred for weight budget" })
